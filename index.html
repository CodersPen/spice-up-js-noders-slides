    return function(b) {
        return a + b;
    }
}

var add1 = add(1);
var add2 = add(2);

add(1)(2); // 3
add1(2);   // 3
add2(3);   // 5

Currying wrapper

                            
function curry(fn) {
    if (fn.length === 0) {
        return fn;
    }

    function nest(N, args) {
        return (x) => {
            if (N - 1 === 0) {
                return fn(...args, x);
            }

        return nest(N - 1, [...args, x]);
        };
    }

    return nest(fn.length, []);
}

const log2 = curry((x, y) => console.log(x, y));
